                  # mysql

                  # 安装

                  -- 初始化MySQL
    mysqld --initialize-insecure

--  注册MySQL服务
    mysqld -install

--   启动MySQL服务
    net start mysql 
--   停止mysql服务
    net stop mysql

--   修改默认账户密码
    mysqladmin -u root password 1234
    -- 登陆
    mysql -uroot -p1234
